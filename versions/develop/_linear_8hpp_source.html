128 template<>
129 inline void Linear<state_representation::CartesianState>::set_attractor(const state_representation::CartesianState& attractor) {
130  if (attractor.is_empty()) {
131  throw state_representation::exceptions::EmptyStateException(attractor.get_name() + " state is empty");
132  }
133  if (this->get_base_frame().is_empty()) {
134  DynamicalSystem<state_representation::CartesianState>::set_base_frame(
135  state_representation::CartesianState::Identity(attractor.get_reference_frame(), attractor.get_reference_frame()));
136  }
137  // validate that the reference frame of the attractor is always compatible with the DS reference frame
138  if (attractor.get_reference_frame() != this->get_base_frame().get_name()) {
139  if (attractor.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
140  throw state_representation::exceptions::IncompatibleReferenceFramesException(
141  "The reference frame of the attractor " + attractor.get_name() + " in frame "
142  + attractor.get_reference_frame() + " is incompatible with the base frame of the dynamical system "
143  + this->get_base_frame().get_name() + " in frame " + this->get_base_frame().get_reference_frame() + ".");
144  }
145  this->attractor_->set_value(this->get_base_frame().inverse() * attractor);
146  } else {
147  this->attractor_->set_value(attractor);
148  }
149 }

164  + this->get_base_frame().get_name() + ".");
