85void CartesianTwistController::set_angular_gains(double angular_stiffness, double angular_damping) {
86 angular_stiffness_->set_value(angular_stiffness);
87 angular_damping_->set_value(angular_damping);
88
89 Eigen::MatrixXd k(6, 6), d(6, 6);
90 k.diagonal() << 0, 0, 0, angular_stiffness, angular_stiffness, angular_stiffness;
91 d.diagonal() << 0, 0, 0, angular_damping, angular_damping, angular_damping;
92 velocity_impedance_ctrl_.set_stiffness(k);
93 velocity_impedance_ctrl_.set_damping(d);
94}
