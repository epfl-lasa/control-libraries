113 JointVelocities JointAccelerations::operator*(const std::chrono::nanoseconds& dt) const {
114  if (this->is_empty()) { throw EmptyStateException(this->get_name() + " state is empty"); }
115  // operations
116  JointVelocities velocities(this->get_name(), this->get_names());
117  // convert the period to a double with the second as reference
118  double period = dt.count();
119  period /= 1e9;
120  // multiply the velocities by this period value and assign it as position
121  velocities.set_velocities(period * this->get_accelerations());
122  return velocities;
123 }
