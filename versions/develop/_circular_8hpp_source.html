179 inline void Circular::set_center(const state_representation::CartesianPose& center) {
180  if (center.is_empty()) {
181  throw state_representation::exceptions::EmptyStateException(center.get_name() + " state is empty");
182  }
183  if (this->get_base_frame().is_empty()) {
184  DynamicalSystem<state_representation::CartesianState>::set_base_frame(state_representation::CartesianState::Identity(
185  center.get_reference_frame(),
186  center.get_reference_frame()));
187  }
188  if (center.get_reference_frame() != this->get_base_frame().get_name()) {
189  if (center.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
190  throw state_representation::exceptions::IncompatibleReferenceFramesException(
191  "The reference frame of the center " + center.get_name() + " in frame " + center.get_reference_frame()
192  + " is incompatible with the base frame of the dynamical system " + this->get_base_frame().get_name()
193  + " in frame " + this->get_base_frame().get_reference_frame() + ".");
194  }
195  this->limit_cycle_->get_value().set_center_state(this->get_base_frame().inverse() * center);
196  } else {
197  this->limit_cycle_->get_value().set_center_state(center);
198  }
199 }
