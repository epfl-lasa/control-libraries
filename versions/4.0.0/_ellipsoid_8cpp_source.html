18 const std::list<CartesianPose> Ellipsoid::sample_from_parameterization(unsigned int nb_samples) const {
19  // use a linespace to have a full rotation angle between [0, 2pi]
20  std::vector<double> alpha = math_tools::linspace(0, 2 * M_PI, nb_samples);
21 
22  std::list<CartesianPose> samples;
23  for (unsigned int i = 0; i < nb_samples; ++i) {
24  CartesianPose point(this->get_name() + "_point" + std::to_string(i), this->get_rotation().get_name());
25  double a = alpha.at(i);
26  Eigen::Vector3d position;
27  position(0) = this->get_axis_length(0) * cos(a);
28  position(1) = this->get_axis_length(1) * sin(a);
29  position(2) = 0;
30  point.set_position(position);
31  samples.push_back(this->get_center_pose() * this->get_rotation() * point);
32  }
33  return samples;
34 }

36 const Ellipsoid Ellipsoid::from_algebraic_equation(const std::string& name,
37  const std::vector<double>& coefficients,
38  const std::string& reference_frame) {
39  // extract all the components from the coefficients vector
40  double b2 = coefficients[1] * coefficients[1];
41  double delta = b2 - 4 * coefficients[0] * coefficients[2];
42 
43  // store intermediate calcualtions
44  double tmp1 = coefficients[0] * coefficients[4] * coefficients[4] // AE2
45  + coefficients[2] * coefficients[3] * coefficients[3] // CD2
46  - coefficients[1] * coefficients[3] * coefficients[4] // BDE
47  + delta * coefficients[5]; // deltaF
48  double tmp2 = coefficients[2] - coefficients[0]; // C-A
49  double tmp3 = sqrt(tmp2 * tmp2 + b2);
50 
51  // create the ellipsoid in the plan and set its center and axis
52  Ellipsoid result(name, reference_frame);
53  std::vector<double> axis_lengths(2);
54 
55  // set axis lengths
56  double r1 = -sqrt(2 * tmp1 * (coefficients[0] + coefficients[2] + tmp3)) / delta;
57  double r2 = -sqrt(2 * tmp1 * (coefficients[0] + coefficients[2] - tmp3)) / delta;
58  axis_lengths[0] = (r1 >= r2) ? r1 : r2;
59  axis_lengths[1] = (r1 >= r2) ? r2 : r1;
60  result.set_axis_lengths(axis_lengths);
61 
62  // set center position
63  double cx = (2 * coefficients[2] * coefficients[3] - coefficients[1] * coefficients[4]) / delta;
64  double cy = (2 * coefficients[0] * coefficients[4] - coefficients[1] * coefficients[3]) / delta;
65  result.set_center_position(Eigen::Vector3d(cx, cy, 0));
66 
67  // set center orientation
68  double phi;
69  if (abs(coefficients[1]) > 1e-4) {
70  phi = atan2(tmp2 - tmp3, coefficients[1]);
71  } else if (coefficients[0] < coefficients[2]) {
72  phi = 0.;
73  } else {
74  phi = M_PI_2;
75  }
76  if (r1 < r2) { phi += M_PI_2; }
77  result.set_rotation_angle(phi);
78 
79  return result;
80 }

82 const Ellipsoid Ellipsoid::fit(const std::string& name,
83  const std::list<CartesianPose>& points,
84  const std::string& reference_frame,
85  double noise_level) {
86  // define the constraint matrix
87  Eigen::SparseMatrix<double> constraint_matrix(6, 6);
88  constraint_matrix.insert(1, 1) = -1;
89  constraint_matrix.insert(0, 2) = 2;
90  constraint_matrix.insert(2, 0) = 2;
91 
92  // preprocessing normalize the data TODO using PCA to find the plan
93  size_t nb_points = points.size();
94  Eigen::VectorXd x_value(nb_points);
95  Eigen::VectorXd y_value(nb_points);
96 
97  // retry with noise everytime delta > 0
98  double delta;
99  std::vector<double> coefficients(6);
100  Eigen::GeneralizedEigenSolver<Eigen::Matrix<double, 6, 6>> solver;
101 
102  std::default_random_engine generator;
103  std::normal_distribution<double> dist(0., noise_level);
104  do {
105  unsigned int i = 0;
106  for (const auto& p : points) {
107  x_value[i] = p.get_position()(0) + dist(generator);
108  y_value[i] = p.get_position()(1) + dist(generator);
109  ++i;
110  }
111 
112  double xm = x_value.mean();
113  double ym = y_value.mean();
114  double sx = 0.5 * (x_value.maxCoeff() - x_value.minCoeff());
115  double sy = 0.5 * (y_value.maxCoeff() - y_value.minCoeff());
116  Eigen::VectorXd x_value_centered = (x_value.array() - xm) / sx;
117  Eigen::VectorXd y_value_centered = (y_value.array() - ym) / sy;
118 
119  // compute the design matrix
120  Eigen::MatrixXd design_matrix(nb_points, 6);
121  design_matrix.col(0) = x_value_centered.array() * x_value_centered.array();
122  design_matrix.col(1) = x_value_centered.array() * y_value_centered.array();
123  design_matrix.col(2) = y_value_centered.array() * y_value_centered.array();
124  design_matrix.col(3) = x_value_centered;
125  design_matrix.col(4) = y_value_centered;
126  design_matrix.col(5) = Eigen::VectorXd::Ones(nb_points);
127 
128  // compute the scatter matrix
129  Eigen::Matrix<double, 6, 6> scatter_matrix = design_matrix.transpose() * design_matrix;
130 
131  // solve the generalized eigenvalue problem
132  solver.compute(scatter_matrix, constraint_matrix);
133 
134  // solution correspon to the single positive eigenvalue
135  unsigned int solution_index;
136  double eigenvalue = solver.betas().maxCoeff(&solution_index);
137 
138  // no solution case
139  if (eigenvalue < 0) {
140  throw exceptions::NoSolutionToFitException("No solution found for the ellipse fitting");
141  }
142 
143  // extract the solution
144  Eigen::VectorXd solution = solver.eigenvectors().col(solution_index).real();
145 
146  // unnormalize the parameters TODO inverse PCA
147  double kx = xm;
148  double ky = ym;
149  double sx2 = sx * sx;
150  double sy2 = sy * sy;
151 
152  coefficients[0] = solution(0) * sy2;
153  coefficients[1] = solution(1) * sx * sy;
154  coefficients[2] = solution(2) * sx2;
155  coefficients[3] = -2 * solution(0) * sy2 * kx - solution(1) * sx * sy * ky + solution(3) * sx * sy2;
156  coefficients[4] = -solution(1) * sx * sy * kx - 2 * solution(2) * sx2 * ky + solution(4) * sx2 * sy;
157  coefficients[5] = solution(0) * sy2 * kx * kx
158  + solution(1) * sx * sy * kx * ky
159  + solution(2) * sx2 * ky * ky
160  - solution(3) * sx * sy2 * kx
161  - solution(4) * sx2 * sy * ky
162  + solution(5) * sx2 * sy2;
163 
164  delta = coefficients[1] * coefficients[1] - 4 * coefficients[0] * coefficients[2];
165  } while (delta > 0);
166  return from_algebraic_equation(name, coefficients, reference_frame);
167 }
