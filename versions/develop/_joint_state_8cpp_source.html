242 void JointState::clamp_state_variable(
243  const Eigen::ArrayXd& max_absolute_value_array, const JointStateVariable& state_variable_type,
244  const Eigen::ArrayXd& noise_ratio_array
245 ) {
246  Eigen::VectorXd state_variable = this->get_state_variable(state_variable_type);
247  int expected_size = state_variable.size();
248  if (max_absolute_value_array.size() != expected_size) {
249  throw IncompatibleSizeException(
250  "Array of max values is of incorrect size: expected " + std::to_string(expected_size) + ", given "
251  + std::to_string(max_absolute_value_array.size()));
252  }
253 
254  if (noise_ratio_array.size() != expected_size) {
255  throw IncompatibleSizeException(
256  "Array of max values is of incorrect size: expected " + std::to_string(expected_size) + ", given "
257  + std::to_string(noise_ratio_array.size()));
258  }
259  for (int i = 0; i < expected_size; ++i) {
260  if (noise_ratio_array(i) != 0.0 && abs(state_variable(i)) < noise_ratio_array(i) * max_absolute_value_array(i)) {
261  // apply dead zone
262  state_variable(i) = 0.0;
263  } else if (abs(state_variable(i)) > max_absolute_value_array(i)) {
264  // clamp to max value
265  state_variable(i) *= max_absolute_value_array(i) / abs(state_variable(i));
266  }
267  }
268  this->set_state_variable(state_variable, state_variable_type);
269 }

281 double JointState::dist(const JointState& state, const JointStateVariable& state_variable_type) const {
282  // sanity check
283  if (this->is_empty()) { throw EmptyStateException(this->get_name() + " state is empty"); }
284  if (state.is_empty()) { throw EmptyStateException(state.get_name() + " state is empty"); }
285  if (!this->is_compatible(state)) {
286  throw IncompatibleStatesException(
287  "The two joint states are incompatible, check name, joint names and order or size"
288  );
289  }
290  // calculation
291  double result = 0;
292  if (state_variable_type == JointStateVariable::POSITIONS || state_variable_type == JointStateVariable::ALL) {
293  result += (this->get_positions() - state.get_positions()).norm();
294  }
295  if (state_variable_type == JointStateVariable::VELOCITIES || state_variable_type == JointStateVariable::ALL) {
296  result += (this->get_velocities() - state.get_velocities()).norm();
297  }
298  if (state_variable_type == JointStateVariable::ACCELERATIONS || state_variable_type == JointStateVariable::ALL) {
299  result += (this->get_accelerations() - state.get_accelerations()).norm();
300  }
301  if (state_variable_type == JointStateVariable::TORQUES || state_variable_type == JointStateVariable::ALL) {
302  result += (this->get_torques() - state.get_torques()).norm();
303  }
304  return result;
305 }
