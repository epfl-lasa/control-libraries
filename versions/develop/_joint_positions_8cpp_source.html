112JointVelocities JointPositions::operator/(const std::chrono::nanoseconds& dt) const {
113 if (this->is_empty()) { throw EmptyStateException(this->get_name() + " state is empty"); }
114 // operations
115 JointVelocities velocities(this->get_name(), this->get_names());
116 // convert the period to a double with the second as reference
117 double period = dt.count();
118 period /= 1e9;
119 // divide the positions by this period value and assign it as velocities
120 velocities.set_velocities(this->get_positions() / period);
121 return velocities;
122}
