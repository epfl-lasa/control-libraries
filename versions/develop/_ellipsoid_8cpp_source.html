20 const std::list<CartesianPose> Ellipsoid::sample_from_parameterization(unsigned int nb_samples) const {
21  if (this->get_center_state().is_empty()) {
22  throw exceptions::EmptyStateException("The center state of the Ellipsoid is not set yet.");
23  }
24  // use a linspace to have a full rotation angle between [0, 2pi]
25  std::vector<double> alpha = math_tools::linspace(0, 2 * M_PI, nb_samples);
26 
27  std::list<CartesianPose> samples;
28  for (unsigned int i = 0; i < nb_samples; ++i) {
29  CartesianPose point(this->get_name() + "_point" + std::to_string(i), this->get_rotation().get_name());
30  double a = alpha.at(i);
31  Eigen::Vector3d position;
32  position(0) = this->get_axis_length(0) * cos(a);
33  position(1) = this->get_axis_length(1) * sin(a);
34  position(2) = 0;
35  point.set_position(position);
36  samples.push_back(this->get_center_pose() * this->get_rotation() * point);
37  }
38  return samples;
39 }

41 const Ellipsoid Ellipsoid::from_algebraic_equation(
42  const std::string& name, const std::vector<double>& coefficients, const std::string& reference_frame
43 ) {
44  // extract all the components from the coefficients vector
45  double b2 = coefficients[1] * coefficients[1];
46  double delta = b2 - 4 * coefficients[0] * coefficients[2];
47 
48  // store intermediate calculations
49  double tmp1 = coefficients[0] * coefficients[4] * coefficients[4] // AE2
50  + coefficients[2] * coefficients[3] * coefficients[3] // CD2
51  - coefficients[1] * coefficients[3] * coefficients[4] // BDE
52  + delta * coefficients[5]; // deltaF
53  double tmp2 = coefficients[2] - coefficients[0]; // C-A
54  double tmp3 = sqrt(tmp2 * tmp2 + b2);
55 
56  // create the ellipsoid in the plan and set its center and axis
57  Ellipsoid result(name, reference_frame);
58  std::vector<double> axis_lengths(2);
59 
60  // set axis lengths
61  double r1 = -sqrt(2 * tmp1 * (coefficients[0] + coefficients[2] + tmp3)) / delta;
62  double r2 = -sqrt(2 * tmp1 * (coefficients[0] + coefficients[2] - tmp3)) / delta;
63  axis_lengths[0] = (r1 >= r2) ? r1 : r2;
64  axis_lengths[1] = (r1 >= r2) ? r2 : r1;
65  result.set_axis_lengths(axis_lengths);
66 
67  // set center position
68  double cx = (2 * coefficients[2] * coefficients[3] - coefficients[1] * coefficients[4]) / delta;
69  double cy = (2 * coefficients[0] * coefficients[4] - coefficients[1] * coefficients[3]) / delta;
70  result.set_center_position(Eigen::Vector3d(cx, cy, 0));
71 
72  // set center orientation
73  double phi;
74  if (abs(coefficients[1]) > 1e-4) {
75  phi = atan2(tmp2 - tmp3, coefficients[1]);
76  } else if (coefficients[0] < coefficients[2]) {
77  phi = 0.;
78  } else {
79  phi = M_PI_2;
80  }
81  if (r1 < r2) { phi += M_PI_2; }
82  result.set_rotation_angle(phi);
83 
84  return result;
85 }

87 const Ellipsoid Ellipsoid::fit(
88  const std::string& name, const std::list<CartesianPose>& points, const std::string& reference_frame,
89  double noise_level
90 ) {
91  // define the constraint matrix
92  Eigen::SparseMatrix<double> constraint_matrix(6, 6);
93  constraint_matrix.insert(1, 1) = -1;
94  constraint_matrix.insert(0, 2) = 2;
95  constraint_matrix.insert(2, 0) = 2;
96 
97  // preprocessing normalize the data TODO using PCA to find the plan
98  size_t nb_points = points.size();
99  Eigen::VectorXd x_value(nb_points);
100  Eigen::VectorXd y_value(nb_points);
101 
102  // retry with noise everytime delta > 0
103  double delta;
104  std::vector<double> coefficients(6);
105  Eigen::GeneralizedEigenSolver<Eigen::Matrix<double, 6, 6>> solver;
106 
107  std::default_random_engine generator;
108  std::normal_distribution<double> dist(0., noise_level);
109  do {
110  unsigned int i = 0;
111  for (const auto& p: points) {
112  x_value[i] = p.get_position()(0) + dist(generator);
113  y_value[i] = p.get_position()(1) + dist(generator);
114  ++i;
115  }
116 
117  double xm = x_value.mean();
118  double ym = y_value.mean();
119  double sx = 0.5 * (x_value.maxCoeff() - x_value.minCoeff());
120  double sy = 0.5 * (y_value.maxCoeff() - y_value.minCoeff());
121  Eigen::VectorXd x_value_centered = (x_value.array() - xm) / sx;
122  Eigen::VectorXd y_value_centered = (y_value.array() - ym) / sy;
123 
124  // compute the design matrix
125  Eigen::MatrixXd design_matrix(nb_points, 6);
126  design_matrix.col(0) = x_value_centered.array() * x_value_centered.array();
127  design_matrix.col(1) = x_value_centered.array() * y_value_centered.array();
128  design_matrix.col(2) = y_value_centered.array() * y_value_centered.array();
129  design_matrix.col(3) = x_value_centered;
130  design_matrix.col(4) = y_value_centered;
131  design_matrix.col(5) = Eigen::VectorXd::Ones(nb_points);
132 
133  // compute the scatter matrix
134  Eigen::Matrix<double, 6, 6> scatter_matrix = design_matrix.transpose() * design_matrix;
135 
136  // solve the generalized eigenvalue problem
137  solver.compute(scatter_matrix, constraint_matrix);
138 
139  // solution correspon to the single positive eigenvalue
140  unsigned int solution_index;
141  double eigenvalue = solver.betas().maxCoeff(&solution_index);
142 
143  // no solution case
144  if (eigenvalue < 0) {
145  throw exceptions::NoSolutionToFitException("No solution found for the ellipse fitting");
146  }
147 
148  // extract the solution
149  Eigen::VectorXd solution = solver.eigenvectors().col(solution_index).real();
150 
151  // unnormalize the parameters TODO inverse PCA
152  double kx = xm;
153  double ky = ym;
154  double sx2 = sx * sx;
155  double sy2 = sy * sy;
156 
157  coefficients[0] = solution(0) * sy2;
158  coefficients[1] = solution(1) * sx * sy;
159  coefficients[2] = solution(2) * sx2;
160  coefficients[3] = -2 * solution(0) * sy2 * kx - solution(1) * sx * sy * ky + solution(3) * sx * sy2;
161  coefficients[4] = -solution(1) * sx * sy * kx - 2 * solution(2) * sx2 * ky + solution(4) * sx2 * sy;
162  coefficients[5] = solution(0) * sy2 * kx * kx + solution(1) * sx * sy * kx * ky + solution(2) * sx2 * ky * ky
163  - solution(3) * sx * sy2 * kx - solution(4) * sx2 * sy * ky + solution(5) * sx2 * sy2;
164 
165  delta = coefficients[1] * coefficients[1] - 4 * coefficients[0] * coefficients[2];
166  } while (delta > 0);
167  return from_algebraic_equation(name, coefficients, reference_frame);
168 }
