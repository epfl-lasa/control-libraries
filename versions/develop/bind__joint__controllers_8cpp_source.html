19py::object joint_controller_factory(
20 CONTROLLER_TYPE type, std::list<std::shared_ptr<ParameterInterface>> parameter_list, unsigned int dimensions = 6
21) {
22 switch (type) {
23 case CONTROLLER_TYPE::IMPEDANCE:
24 return py::cast(impedance::Impedance<JointState>(parameter_list, dimensions));
25 case CONTROLLER_TYPE::VELOCITY_IMPEDANCE:
26 return py::cast(impedance::VelocityImpedance<JointState>(parameter_list, dimensions));
27 case CONTROLLER_TYPE::DISSIPATIVE:
28 return py::cast(
29 impedance::Dissipative<JointState>(parameter_list, impedance::ComputationalSpaceType::FULL, dimensions));
30 case CONTROLLER_TYPE::DISSIPATIVE_LINEAR:
31 case CONTROLLER_TYPE::DISSIPATIVE_ANGULAR:
32 case CONTROLLER_TYPE::DISSIPATIVE_DECOUPLED:
33 throw controllers::exceptions::InvalidControllerException(
34 "The JointState dissipative controller cannot be decoupled into linear and angular terms."
35 );
36 case CONTROLLER_TYPE::COMPLIANT_TWIST:
37 throw controllers::exceptions::InvalidControllerException(
38 "The compliant twist controller only works in Cartesian space."
39 );
40 default:
41 case CONTROLLER_TYPE::NONE:
42 return py::cast(nullptr);
43 }
44}

46py::object joint_controller_factory(
47 CONTROLLER_TYPE type, std::list<std::shared_ptr<ParameterInterface>> parameter_list,
48 const robot_model::Model& robot_model
49) {
50 switch (type) {
51 case CONTROLLER_TYPE::IMPEDANCE: {
52 auto ctrl = impedance::Impedance<JointState>(parameter_list, robot_model.get_number_of_joints());
53 ctrl.set_robot_model(robot_model);
54 return py::cast(ctrl);
55 }
56 case CONTROLLER_TYPE::VELOCITY_IMPEDANCE: {
57 auto ctrl = impedance::VelocityImpedance<JointState>(parameter_list, robot_model.get_number_of_joints());
58 ctrl.set_robot_model(robot_model);
59 return py::cast(ctrl);
60 }
61 case CONTROLLER_TYPE::DISSIPATIVE: {
62 auto ctrl = impedance::Dissipative<JointState>(
63 parameter_list, impedance::ComputationalSpaceType::FULL, robot_model.get_number_of_joints());
64 ctrl.set_robot_model(robot_model);
65 return py::cast(ctrl);
66 }
67 case CONTROLLER_TYPE::DISSIPATIVE_LINEAR:
68 case CONTROLLER_TYPE::DISSIPATIVE_ANGULAR:
69 case CONTROLLER_TYPE::DISSIPATIVE_DECOUPLED:
70 throw controllers::exceptions::InvalidControllerException(
71 "The JointState dissipative controller cannot be decoupled into linear and angular terms."
72 );
73 case CONTROLLER_TYPE::COMPLIANT_TWIST:
74 throw controllers::exceptions::InvalidControllerException(
75 "The compliant twist controller only works in Cartesian space."
76 );
77 default:
78 case CONTROLLER_TYPE::NONE:
79 throw controllers::exceptions::InvalidControllerException("Cannot assign robot model to this controller!");
80 }
81}
