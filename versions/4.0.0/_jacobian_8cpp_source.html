88 bool Jacobian::is_compatible(const State& state) const {
89  bool compatible = false;
90  switch (state.get_type()) {
91  case StateType::JACOBIANMATRIX:
92  // compatibility is assured through the vector of joint names
93  compatible = (this->get_name() == state.get_name())
94  && (this->cols_ == dynamic_cast<const Jacobian&>(state).get_joint_names().size());
95  if (compatible) {
96  for (unsigned int i = 0; i < this->cols_; ++i) {
97  compatible = (compatible && this->joint_names_[i] == dynamic_cast<const Jacobian&>(state).get_joint_names()[i]);
98  }
99  // compatibility is assured through the reference frame and the name of the frame
100  compatible = (compatible && ((this->reference_frame_ == dynamic_cast<const Jacobian&>(state).get_reference_frame())
101  && (this->frame_ == dynamic_cast<const Jacobian&>(state).get_frame())));
102  }
103  break;
104  case StateType::JOINTSTATE:
105  // compatibility is assured through the vector of joint names
106  compatible = (this->get_name() == state.get_name())
107  && (this->cols_ == dynamic_cast<const JointState&>(state).get_size());
108  if (compatible) {
109  for (unsigned int i = 0; i < this->cols_; ++i) {
110  compatible = (compatible && this->joint_names_[i] == dynamic_cast<const JointState&>(state).get_names()[i]);
111  }
112  }
113  break;
114  case StateType::CARTESIANSTATE:
115  // compatibility is assured through the reference frame and the name of the frame
116  compatible = (this->reference_frame_ == dynamic_cast<const CartesianState&>(state).get_reference_frame())
117  && (this->frame_ == dynamic_cast<const CartesianState&>(state).get_name());
118  break;
119  default:
120  break;
121  }
122  return compatible;
123 }

277 Jacobian operator*(const CartesianPose& pose, const Jacobian& jacobian) {
278  // check compatibility
279  if (jacobian.is_empty()) {
280  throw EmptyStateException(jacobian.get_name() + " state is empty");
281  }
282  if (pose.is_empty()) {
283  throw EmptyStateException(pose.get_name() + " state is empty");
284  }
285  if (pose.get_name() != jacobian.get_reference_frame()) {
286  throw IncompatibleStatesException("The Jacobian and the input CartesianPose are incompatible, expected pose of "
287  + jacobian.get_reference_frame() + " got " + pose.get_name());
288  }
289  // number of rows of the jacobian should be 6 (incorrect if it has been transposed before)
290  if (jacobian.rows_ != 6) {
291  throw IncompatibleStatesException(
292  "The Jacobian and the input CartesianPose are incompatible, the Jacobian has probably been transposed before");
293  }
294  Jacobian result(jacobian);
295  // change the reference frame of all the columns
296  for (unsigned int i = 0; i < jacobian.cols_; ++i) {
297  // update position part
298  result.data_.col(i).head(3) = pose.get_orientation() * jacobian.data_.col(i).head(3);
299  // update orientation part
300  result.data_.col(i).tail(3) = pose.get_orientation() * jacobian.data_.col(i).tail(3);
301  }
302  // change the reference frame
303  result.reference_frame_ = pose.get_reference_frame();
304  return result;
305 }
