36 CartesianState Circular::compute_dynamics(const CartesianState& state) const {
37  if (this->get_limit_cycle().get_center_state().is_empty()) {
38  throw exceptions::EmptyAttractorException("The limit cycle of the dynamical system is empty.");
39  }
40  // put the point in the reference of the center
41  auto pose = CartesianPose(state);
42  pose = this->get_limit_cycle().get_rotation().inverse() * this->get_center().inverse() * pose;
43 
44  CartesianTwist velocity(pose.get_name(), pose.get_reference_frame());
45  Eigen::Vector3d linear_velocity;
46  linear_velocity(2) = -this->get_normal_gain() * pose.get_position()(2);
47 
48  std::vector<double> radiuses = this->get_radiuses();
49 
50  double a2ratio = (pose.get_position()[0] * pose.get_position()[0]) / (radiuses[0] * radiuses[0]);
51  double b2ratio = (pose.get_position()[1] * pose.get_position()[1]) / (radiuses[1] * radiuses[1]);
52  double dradius = -this->get_planar_gain() * radiuses[0] * radiuses[1] * (a2ratio + b2ratio - 1);
53  double tangent_velocity_x = -radiuses[0] / radiuses[1] * pose.get_position()[1];
54  double tangent_velocity_y = radiuses[1] / radiuses[0] * pose.get_position()[0];
55 
56  linear_velocity(0) = this->get_circular_velocity() * tangent_velocity_x + dradius * tangent_velocity_y;
57  linear_velocity(1) = this->get_circular_velocity() * tangent_velocity_y - dradius * tangent_velocity_x;
58 
59  velocity.set_linear_velocity(linear_velocity);
60  velocity.set_angular_velocity(Eigen::Vector3d::Zero());
61 
62  //compute back the linear velocity in the desired frame
63  auto frame = this->get_center() * this->get_limit_cycle().get_rotation();
64  return CartesianState(frame) * velocity;
65 }
