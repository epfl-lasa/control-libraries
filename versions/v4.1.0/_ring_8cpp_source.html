46 CartesianState Ring::compute_dynamics(const CartesianState& state) const {
47  if (this->get_center().is_empty()) {
48  throw exceptions::EmptyAttractorException("The center of the dynamical system is empty.");
49  }
50  // put the point in the reference of the center
51  CartesianPose pose(state);
52  pose = this->get_center().inverse() * pose;
53  // apply the rotation offset
54  pose.set_orientation(pose.get_orientation() * this->get_rotation_offset().conjugate());
55 
56  CartesianTwist twist(pose.get_name(), pose.get_reference_frame());
57  double local_field_strength;
58  twist.set_linear_velocity(this->calculate_local_linear_velocity(pose, local_field_strength));
59  twist.set_angular_velocity(this->calculate_local_angular_velocity(pose,
60  twist.get_linear_velocity(),
61  local_field_strength));
62 
63  // transform the twist back to the base reference frame
64  return CartesianState(this->get_center()) * twist;
65 }

67 Eigen::Vector3d Ring::calculate_local_linear_velocity(const CartesianPose& pose, double& local_field_strength) const {
68  Eigen::Vector3d local_linear_velocity = Eigen::Vector3d::Zero();
69 
70  // get the 2d components of position on the XY plane
71  Eigen::Vector2d position2d(pose.get_position().x(), pose.get_position().y());
72 
73  double d = position2d.norm();
74  if (d < 1e-7) {
75  return local_linear_velocity;
76  }
77 
78  double re = M_PI_2 * (d - this->get_radius()) / this->get_width();
79  if (re > M_PI_2) {
80  re = M_PI_2;
81  } else if (re < -M_PI_2) {
82  re = -M_PI_2;
83  }
84 
85  // calculate the velocity of a point as an orthogonal unit vector, rectified towards the radius based on re
86  Eigen::Matrix2d R;
87  R << -sin(re), -cos(re), cos(re), -sin(re);
88  Eigen::Vector2d velocity2d = R * position2d / d;
89 
90  // scale by the nominal speed
91  velocity2d *= this->get_speed();
92 
93  // calculate the normal velocity
94  double vz = -this->get_normal_gain() * pose.get_position().z();
95 
96  // combine into 3D velocity
97  local_linear_velocity << velocity2d, vz;
98 
99  // calculate the field strength and scale the velocity
100  local_field_strength = this->get_field_strength() + (1 - this->get_field_strength()) * cos(re);
101  local_linear_velocity *= local_field_strength;
102 
103  return local_linear_velocity;
104 }

106 Eigen::Vector3d Ring::calculate_local_angular_velocity(const CartesianPose& pose,
107  const Eigen::Vector3d& linearVelocity,
108  double local_field_strength) const {
109  Eigen::Vector3d local_angular_velocity = Eigen::Vector3d::Zero();
110 
111  double theta = atan2(pose.get_position().y(), pose.get_position().x());
112 
113  Eigen::Quaterniond qd = Eigen::Quaterniond::Identity();
114  qd.w() = cos(theta / 2);
115  qd.z() = sin(theta / 2);
116 
117  if (pose.get_orientation().dot(qd) < 0) {
118  qd.coeffs() = -qd.coeffs();
119  }
120 
121  Eigen::Quaterniond deltaQ = qd * pose.get_orientation().conjugate();
122  if (deltaQ.vec().norm() < 1e-7) {
123  return local_angular_velocity;
124  }
125 
126  //dOmega = 2 * ln (deltaQ)
127  Eigen::Quaterniond deltaOmega = Eigen::Quaterniond::Identity();
128  deltaOmega.w() = 0;
129  double phi = atan2(deltaQ.vec().norm(), deltaQ.w());
130  deltaOmega.vec() = 2 * deltaQ.vec() * phi / sin(phi);
131 
132  local_angular_velocity = get_angular_gain() * deltaOmega.vec();
133  local_angular_velocity *= local_field_strength;
134 
135  Eigen::Vector2d position2d(pose.get_position().x(), pose.get_position().y());
136  Eigen::Vector2d linear_velocity2d(linearVelocity.x(), linearVelocity.y());
137  if (position2d.norm() < 1e-7 || linear_velocity2d.norm() < 1e-7) {
138  return local_angular_velocity;
139  }
140 
141  // clamp linear velocity magnitude to half the position, so that the angular displacement
142  // around the local Z axis caused by the linear velocity stays within one quadrant
143  if (linear_velocity2d.norm() > (0.5 * position2d.norm())) {
144  linear_velocity2d = linear_velocity2d.normalized() * (0.5 * position2d).norm();
145  }
146  double projection = position2d.normalized().dot((position2d + linear_velocity2d).normalized());
147  double dthetaZ = 0;
148  if (1 - abs(projection) > 1e-7) {
149  dthetaZ = acos(projection);
150  }
151  local_angular_velocity.z() += dthetaZ;
152 
153  return local_angular_velocity;
154 }

169 void Ring::set_center(const CartesianPose& center) {
170  if (center.is_empty()) {
171  throw state_representation::exceptions::EmptyStateException(center.get_name() + " state is empty");
172  }
173  if (this->get_base_frame().is_empty()) {
174  DynamicalSystem<CartesianState>::set_base_frame(CartesianState::Identity(center.get_reference_frame(),
175  center.get_reference_frame()));
176  }
177  // validate that the reference frame of the center is always compatible with the DS reference frame
178  if (center.get_reference_frame() != this->get_base_frame().get_name()) {
179  if (center.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
180  throw state_representation::exceptions::IncompatibleReferenceFramesException(
181  "The reference frame of the center " + center.get_name() + " in frame " + center.get_reference_frame()
182  + " is incompatible with the base frame of the dynamical system " + this->get_base_frame().get_name()
183  + " in frame " + this->get_base_frame().get_reference_frame() + ".");
184  }
185  this->center_->set_value(this->get_base_frame().inverse() * center);
186  } else {
187  this->center_->set_value(center);
188  }
189  this->set_rotation_offset(this->get_rotation_offset());
190 }
