114JointAccelerations JointVelocities::operator/(const std::chrono::nanoseconds& dt) const {
115 if (this->is_empty()) { throw EmptyStateException(this->get_name() + " state is empty"); }
116 // operations
117 JointAccelerations accelerations(this->get_name(), this->get_names());
118 // convert the period to a double with the second as reference
119 double period = dt.count();
120 period /= 1e9;
121 // divide the positions by this period value and assign it as velocities
122 accelerations.set_accelerations(this->get_velocities() / period);
123 return accelerations;
124}

126JointPositions JointVelocities::operator*(const std::chrono::nanoseconds& dt) const {
127 if (this->is_empty()) { throw EmptyStateException(this->get_name() + " state is empty"); }
128 // operations
129 JointPositions displacement(this->get_name(), this->get_names());
130 // convert the period to a double with the second as reference
131 double period = dt.count();
132 period /= 1e9;
133 // multiply the velocities by this period value and assign it as position
134 displacement.set_positions(period * this->get_velocities());
135 return displacement;
136}
