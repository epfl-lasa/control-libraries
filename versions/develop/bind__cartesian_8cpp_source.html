70 m.def("create_cartesian_ds", [](DynamicalSystemFactory<CartesianState>::DYNAMICAL_SYSTEM type) -> py::object {
71 switch (type) {
72 case DynamicalSystemFactory<CartesianState>::DYNAMICAL_SYSTEM::CIRCULAR:
73 return py::cast(Circular());
74 break;
75 case DynamicalSystemFactory<CartesianState>::DYNAMICAL_SYSTEM::POINT_ATTRACTOR:
76 return py::cast(PointAttractor<CartesianState>());

78 case DynamicalSystemFactory<CartesianState>::DYNAMICAL_SYSTEM::RING:
79 return py::cast(Ring());
80 break;
81 default:
