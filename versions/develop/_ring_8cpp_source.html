35 void Ring::set_center(const CartesianPose& center) {
36  if (center.is_empty()) {
37  throw state_representation::exceptions::EmptyStateException(center.get_name() + " state is empty");
38  }
39  if (this->get_base_frame().is_empty()) {
40  IDynamicalSystem<CartesianState>::set_base_frame(
41  CartesianState::Identity(center.get_reference_frame(), center.get_reference_frame()));
42  }
43  // validate that the reference frame of the center is always compatible with the DS reference frame
44  if (center.get_reference_frame() != this->get_base_frame().get_name()) {
45  if (center.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
46  throw state_representation::exceptions::IncompatibleReferenceFramesException(
47  "The reference frame of the center " + center.get_name() + " in frame " + center.get_reference_frame()
48  + " is incompatible with the base frame of the dynamical system " + this->get_base_frame().get_name()
49  + " in frame " + this->get_base_frame().get_reference_frame() + "."
50  );
51  }
52  this->center_->set_value(this->get_base_frame().inverse() * center);
53  } else {
54  this->center_->set_value(center);
55  }
56  this->set_rotation_offset(this->get_rotation_offset());

111 Eigen::Vector3d Ring::calculate_local_linear_velocity(const CartesianPose& pose, double& local_field_strength) const {
112  Eigen::Vector3d local_linear_velocity = Eigen::Vector3d::Zero();
113 
114  // get the 2d components of position on the XY plane
115  Eigen::Vector2d position2d(pose.get_position().x(), pose.get_position().y());
116 
117  double d = position2d.norm();
118  if (d < 1e-7) {
119  return local_linear_velocity;
120  }
121 
122  double re = M_PI_2 * (d - this->radius_->get_value()) / this->width_->get_value();
123  if (re > M_PI_2) {
124  re = M_PI_2;
125  } else if (re < -M_PI_2) {
126  re = -M_PI_2;
127  }
128 
129  // calculate the velocity of a point as an orthogonal unit vector, rectified towards the radius based on re
130  Eigen::Matrix2d R;
131  R << -sin(re), -cos(re), cos(re), -sin(re);
132  Eigen::Vector2d velocity2d = R * position2d / d;
133 
134  // scale by the nominal speed
135  velocity2d *= this->speed_->get_value();
136 
137  // calculate the normal velocity
138  double vz = -this->normal_gain_->get_value() * pose.get_position().z();
139 
140  // combine into 3D velocity
141  local_linear_velocity << velocity2d, vz;
142 
143  // calculate the field strength and scale the velocity
144  local_field_strength = this->field_strength_->get_value() + (1 - this->field_strength_->get_value()) * cos(re);
145  local_linear_velocity *= local_field_strength;
146 
147  return local_linear_velocity;
148 }

150 Eigen::Vector3d Ring::calculate_local_angular_velocity(
151  const CartesianPose& pose, const Eigen::Vector3d& linearVelocity, double local_field_strength
152 ) const {
153  Eigen::Vector3d local_angular_velocity = Eigen::Vector3d::Zero();
154 
155  double theta = atan2(pose.get_position().y(), pose.get_position().x());
156 
157  Eigen::Quaterniond qd = Eigen::Quaterniond::Identity();
158  qd.w() = cos(theta / 2);
159  qd.z() = sin(theta / 2);
160 
161  if (pose.get_orientation().dot(qd) < 0) {
162  qd.coeffs() = -qd.coeffs();
163  }
164 
165  Eigen::Quaterniond deltaQ = qd * pose.get_orientation().conjugate();
166  if (deltaQ.vec().norm() < 1e-7) {
167  return local_angular_velocity;
168  }
169 
170  //dOmega = 2 * ln (deltaQ)
171  Eigen::Quaterniond deltaOmega = Eigen::Quaterniond::Identity();
172  deltaOmega.w() = 0;
173  double phi = atan2(deltaQ.vec().norm(), deltaQ.w());
174  deltaOmega.vec() = 2 * deltaQ.vec() * phi / sin(phi);
175 
176  local_angular_velocity = this->angular_gain_->get_value() * deltaOmega.vec();
177  local_angular_velocity *= local_field_strength;
178 
179  Eigen::Vector2d position2d(pose.get_position().x(), pose.get_position().y());
180  Eigen::Vector2d linear_velocity2d(linearVelocity.x(), linearVelocity.y());
181  if (position2d.norm() < 1e-7 || linear_velocity2d.norm() < 1e-7) {
182  return local_angular_velocity;
183  }
184 
185  // clamp linear velocity magnitude to half the position, so that the angular displacement
186  // around the local Z axis caused by the linear velocity stays within one quadrant
187  if (linear_velocity2d.norm() > (0.5 * position2d.norm())) {
188  linear_velocity2d = linear_velocity2d.normalized() * (0.5 * position2d).norm();
189  }
190  double projection = position2d.normalized().dot((position2d + linear_velocity2d).normalized());
191  double dthetaZ = 0;
192  if (1 - abs(projection) > 1e-7) {
193  dthetaZ = acos(projection);
194  }
195  local_angular_velocity.z() += dthetaZ;
196 
197  return local_angular_velocity;
198 }

200 CartesianState Ring::compute_dynamics(const CartesianState& state) const {
201  if (this->center_->get_value().is_empty()) {
202  throw exceptions::EmptyAttractorException("The center of the dynamical system is empty.");
203  }
204  // put the point in the reference of the center
205  CartesianPose pose(state);
206  pose = this->center_->get_value().inverse() * pose;
207  // apply the rotation offset
208  pose.set_orientation(pose.get_orientation() * this->get_rotation_offset().conjugate());
209 
210  CartesianTwist twist(pose.get_name(), pose.get_reference_frame());
211  double local_field_strength;
212  twist.set_linear_velocity(this->calculate_local_linear_velocity(pose, local_field_strength));
213  twist.set_angular_velocity(
214  this->calculate_local_angular_velocity(
215  pose, twist.get_linear_velocity(), local_field_strength
216  ));
217 
218  // transform the twist back to the base reference frame
219  return CartesianState(this->center_->get_value()) * twist;
220 }
