520inline void swap(CartesianState& state1, CartesianState& state2) {
521 swap(static_cast<SpatialState&>(state1), static_cast<SpatialState&>(state2));
522 std::swap(state1.position_, state2.position_);
523 std::swap(state1.orientation_, state2.orientation_);
524 std::swap(state1.linear_velocity_, state2.linear_velocity_);
525 std::swap(state1.angular_velocity_, state2.angular_velocity_);
526 std::swap(state1.linear_acceleration_, state2.linear_acceleration_);
527 std::swap(state1.angular_acceleration_, state2.angular_acceleration_);
528 std::swap(state1.force_, state2.force_);
529 std::swap(state1.torque_, state2.torque_);
530}

606inline Eigen::VectorXd CartesianState::get_state_variable(const CartesianStateVariable& state_variable_type) const {
607 switch (state_variable_type) {
608 case CartesianStateVariable::POSITION:
609 return this->get_position();
610
611 case CartesianStateVariable::ORIENTATION:
612 return this->get_orientation_coefficients();
613
614 case CartesianStateVariable::POSE:
615 return this->get_pose();
616
617 case CartesianStateVariable::LINEAR_VELOCITY:
618 return this->get_linear_velocity();
619
620 case CartesianStateVariable::ANGULAR_VELOCITY:
621 return this->get_angular_velocity();
622
623 case CartesianStateVariable::TWIST:
624 return this->get_twist();
625
626 case CartesianStateVariable::LINEAR_ACCELERATION:
627 return this->get_linear_acceleration();
628
629 case CartesianStateVariable::ANGULAR_ACCELERATION:
630 return this->get_angular_acceleration();
631
632 case CartesianStateVariable::ACCELERATION:
633 return this->get_acceleration();
634
635 case CartesianStateVariable::FORCE:
636 return this->get_force();
637
638 case CartesianStateVariable::TORQUE:
639 return this->get_torque();
640
641 case CartesianStateVariable::WRENCH:
642 return this->get_wrench();
643
644 case CartesianStateVariable::ALL:
645 Eigen::VectorXd all_fields(25);
646 all_fields << this->get_pose(), this->get_twist(), this->get_acceleration(), this->get_wrench();
647 return all_fields;
648 }
649 // this never goes here but is compulsory to avoid a warning
650 return Eigen::Vector3d::Zero();
651}
