42 bool PointAttractor<JointState>::is_compatible(const JointState& state) const {
43  auto attractor = this->get_parameter_value<JointPositions>("attractor");
44  if (attractor.is_empty()) {
45  throw exceptions::EmptyAttractorException("The attractor of the dynamical system is empty.");
46  }
47  bool compatible = (attractor.get_size() == state.get_size());
48  if (compatible) {
49  for (unsigned int i = 0; i < attractor.get_size(); ++i) {
50  compatible = (compatible && attractor.get_names()[i] == state.get_names()[i]);
51  }
52  }
53  return compatible;
54 }

56 template<class S>
57 void PointAttractor<S>::set_gain(const std::shared_ptr<ParameterInterface>& parameter, unsigned int expected_size) {
58  if (parameter->get_type() == StateType::PARAMETER_DOUBLE) {
59  auto gain = std::static_pointer_cast<Parameter<double>>(parameter);
60  this->gain_->set_value(gain->get_value() * Eigen::MatrixXd::Identity(expected_size, expected_size));
61  } else if (parameter->get_type() == StateType::PARAMETER_DOUBLE_ARRAY) {
62  auto gain = std::static_pointer_cast<Parameter<std::vector<double>>>(parameter);
63  if (gain->get_value().size() != expected_size) {
64  throw exceptions::IncompatibleSizeException(
65  "The provided diagonal coefficients do not correspond to the expected size of the attractor"
66  );
67  }
68  Eigen::VectorXd diagonal = Eigen::VectorXd::Map(gain->get_value().data(), expected_size);
69  this->gain_->set_value(diagonal.asDiagonal());
70  } else if (parameter->get_type() == StateType::PARAMETER_MATRIX) {
71  auto gain = std::static_pointer_cast<Parameter<Eigen::MatrixXd>>(parameter);
72  if (gain->get_value().rows() != expected_size && gain->get_value().cols() != expected_size) {
73  throw exceptions::IncompatibleSizeException(
74  "The provided gain matrix do not have the expected size (" + std::to_string(expected_size) + "x"
75  + std::to_string(expected_size) + ")"
76  );
77  }
78  this->gain_->set_value(gain->get_value());
79  } else {
80  throw exceptions::InvalidParameterException("Parameter 'gain' has incorrect type");
81  }
82 }

89 template<>
90 void PointAttractor<CartesianState>::set_attractor(const CartesianState& attractor) {
91  if (attractor.is_empty()) {
92  throw state_representation::exceptions::EmptyStateException(attractor.get_name() + " state is empty");
93  }
94  if (this->get_base_frame().is_empty()) {
95  IDynamicalSystem<CartesianState>::set_base_frame(
96  CartesianState::Identity(attractor.get_reference_frame(), attractor.get_reference_frame()));
97  }
98  // validate that the reference frame of the attractor is always compatible with the DS reference frame
99  if (attractor.get_reference_frame() != this->get_base_frame().get_name()) {
100  if (attractor.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
101  throw state_representation::exceptions::IncompatibleReferenceFramesException(
102  "The reference frame of the attractor " + attractor.get_name() + " in frame "
103  + attractor.get_reference_frame() + " is incompatible with the base frame of the dynamical system "
104  + this->get_base_frame().get_name() + " in frame " + this->get_base_frame().get_reference_frame() + "."
105  );
106  }
107  this->attractor_->set_value(this->get_base_frame().inverse() * attractor);
108  } else {
109  this->attractor_->set_value(attractor);
110  }
111 }

172 template<>
173 CartesianState PointAttractor<CartesianState>::compute_dynamics(const CartesianState& state) const {
174  if (this->attractor_->get_value().is_empty()) {
175  throw exceptions::EmptyAttractorException("The attractor of the dynamical system is empty.");
176  }
177  CartesianTwist twist = CartesianPose(this->attractor_->get_value()) - CartesianPose(state);
178  twist *= this->gain_->get_value();
179  return CartesianTwist(state.get_name(), twist.get_twist(), this->attractor_->get_value().get_reference_frame());
180 }
