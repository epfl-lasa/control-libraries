115 CartesianTwist CartesianPose::operator/(const std::chrono::nanoseconds& dt) const {
116  // sanity check
117  if (this->is_empty()) {
118  throw EmptyStateException(this->get_name() + " state is empty");
119  }
120  // operations
121  CartesianTwist twist(this->get_name(), this->get_reference_frame());
122  // convert the period to a double with the second as reference
123  double period = dt.count();
124  period /= 1e9;
125  // set linear velocity
126  twist.set_linear_velocity(this->get_position() / period);
127  // set angular velocity from the log of the quaternion error
128  Eigen::Quaterniond log_q = math_tools::log(this->get_orientation());
129  if (this->get_orientation().dot(log_q) < 0) {
130  log_q = Eigen::Quaterniond(-log_q.coeffs());
131  }
132  twist.set_angular_velocity(2 * log_q.vec() / period);
133  return twist;
134 }
