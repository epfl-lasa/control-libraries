104template<class S>
105Eigen::MatrixXd Impedance<S>::gain_matrix_from_parameter(
106 const std::shared_ptr<state_representation::ParameterInterface>& parameter
107) {
108 Eigen::MatrixXd matrix;
109 if (parameter->get_type() == state_representation::StateType::PARAMETER_DOUBLE) {
110 auto gain = std::static_pointer_cast<state_representation::Parameter<double>>(parameter);
111 matrix = gain->get_value() * Eigen::MatrixXd::Identity(this->dimensions_, this->dimensions_);
112 } else if (parameter->get_type() == state_representation::StateType::PARAMETER_DOUBLE_ARRAY) {
113 auto gain = std::static_pointer_cast<state_representation::Parameter<std::vector<double>>>(parameter);
114 if (gain->get_value().size() != this->dimensions_) {
115 throw state_representation::exceptions::IncompatibleSizeException(
116 "The provided diagonal coefficients do not match the dimensionality of the controller ("
117 + std::to_string(this->dimensions_) + ")");
118 }
119 Eigen::VectorXd diagonal = Eigen::VectorXd::Map(gain->get_value().data(), this->dimensions_);
120 matrix = diagonal.asDiagonal();
121 } else if (parameter->get_type() == state_representation::StateType::PARAMETER_VECTOR) {
122 auto gain = std::static_pointer_cast<state_representation::Parameter<Eigen::VectorXd>>(parameter);
123 if (gain->get_value().size() != this->dimensions_) {
124 throw state_representation::exceptions::IncompatibleSizeException(
125 "The provided diagonal coefficients do not match the dimensionality of the controller ("
126 + std::to_string(this->dimensions_) + ")");
127 }
128 Eigen::VectorXd diagonal = gain->get_value();
129 matrix = diagonal.asDiagonal();
130 } else if (parameter->get_type() == state_representation::StateType::PARAMETER_MATRIX) {
131 auto gain = std::static_pointer_cast<state_representation::Parameter<Eigen::MatrixXd>>(parameter);
132 if (gain->get_value().rows() != this->dimensions_ || gain->get_value().cols() != this->dimensions_) {
133 auto dim = std::to_string(this->dimensions_);
134 throw state_representation::exceptions::IncompatibleSizeException(
135 "The provided matrix does not have the expected size (" + dim + "x" + dim + ")");
136 }
137 matrix = gain->get_value();
138 } else {
139 throw state_representation::exceptions::InvalidParameterException(
140 "Parameter " + parameter->get_name() + " has incorrect type");
141 }
142 return matrix;
143}
