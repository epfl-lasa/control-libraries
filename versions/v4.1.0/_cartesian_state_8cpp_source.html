48 CartesianState& CartesianState::operator*=(double lambda) {
49  // sanity check
50  if (this->is_empty()) {
51  throw EmptyStateException(this->get_name() + " state is empty");
52  }
53  // operation
54  this->set_position(lambda * this->get_position());
55  // calculate the scaled rotation as a displacement from identity
56  Eigen::Quaterniond w = math_tools::log(this->get_orientation());
57  // calculate the orientation corresponding to the scaled velocity
58  this->set_orientation(math_tools::exp(w, lambda / 2.));
59  // calculate the other vectors normally
60  this->set_twist(lambda * this->get_twist());
61  this->set_accelerations(lambda * this->get_accelerations());
62  this->set_wrench(lambda * this->get_wrench());
63  return (*this);
64 }

107 CartesianState& CartesianState::operator*=(const CartesianState& state) {
108  // sanity check
109  if (this->is_empty()) {
110  throw EmptyStateException(this->get_name() + " state is empty");
111  }
112  if (state.is_empty()) {
113  throw EmptyStateException(state.get_name() + " state is empty");
114  }
115  if (this->get_name() != state.get_reference_frame()) {
116  throw IncompatibleReferenceFramesException("Expected " + this->get_name() + ", got " + state.get_reference_frame());
117  }
118  this->set_name(state.get_name());
119  // intermediate variables for f_S_b
120  Eigen::Vector3d f_P_b = this->get_position();
121  Eigen::Quaterniond f_R_b = this->get_orientation();
122  Eigen::Vector3d f_v_b = this->get_linear_velocity();
123  Eigen::Vector3d f_omega_b = this->get_angular_velocity();
124  Eigen::Vector3d f_a_b = this->get_linear_acceleration();
125  Eigen::Vector3d f_alpha_b = this->get_angular_acceleration();
126  // intermediate variables for b_S_c
127  Eigen::Vector3d b_P_c = state.get_position();
128  Eigen::Quaterniond b_R_c = (this->get_orientation().dot(state.get_orientation()) > 0) ? state.get_orientation()
129  : Eigen::Quaterniond(-state.get_orientation().coeffs());
130  Eigen::Vector3d b_v_c = state.get_linear_velocity();
131  Eigen::Vector3d b_omega_c = state.get_angular_velocity();
132  Eigen::Vector3d b_a_c = state.get_linear_acceleration();
133  Eigen::Vector3d b_alpha_c = state.get_angular_acceleration();
134  // pose
135  this->set_position(f_P_b + f_R_b * b_P_c);
136  this->set_orientation(f_R_b * b_R_c);
137  // twist
138  this->set_linear_velocity(f_v_b + f_R_b * b_v_c + f_omega_b.cross(f_R_b * b_P_c));
139  this->set_angular_velocity(f_omega_b + f_R_b * b_omega_c);
140  // acceleration
141  this->set_linear_acceleration(f_a_b + f_R_b * b_a_c
142  + f_alpha_b.cross(f_R_b * b_P_c)
143  + 2 * f_omega_b.cross(f_R_b * b_v_c)
144  + f_omega_b.cross(f_omega_b.cross(f_R_b * b_P_c)));
145  this->set_angular_acceleration(f_alpha_b + f_R_b * b_alpha_c + f_omega_b.cross(f_R_b * b_omega_c));
146  // wrench
147  //TODO
148  return (*this);
149 }

241  // wrench

253 void CartesianState::clamp_state_variable(
254  double max_norm, const CartesianStateVariable& state_variable_type, double noise_ratio
255 ) {
256  if (state_variable_type == CartesianStateVariable::ORIENTATION || state_variable_type == CartesianStateVariable::POSE
257  || state_variable_type == CartesianStateVariable::ALL) {
258  throw NotImplementedException("clamp_state_variable is not implemented for this CartesianStateVariable");
259  }
260  Eigen::VectorXd state_variable_value = this->get_state_variable(state_variable_type);
261  if (noise_ratio != 0 && state_variable_value.norm() < noise_ratio * max_norm) {
262  // apply a dead zone
263  state_variable_value.setZero();
264  } else if (state_variable_value.norm() > max_norm) {
265  // clamp the values to their maximum amplitude provided
266  state_variable_value = max_norm * state_variable_value.normalized();
267  }
268  this->set_state_variable(state_variable_value, state_variable_type);
269 }

271 double CartesianState::dist(const CartesianState& state, const CartesianStateVariable& state_variable_type) const {
272  // sanity check
273  if (this->is_empty()) {
274  throw EmptyStateException(this->get_name() + " state is empty");
275  }
276  if (state.is_empty()) {
277  throw EmptyStateException(state.get_name() + " state is empty");
278  }
279  if (!(this->get_reference_frame() == state.get_reference_frame())) {
280  throw IncompatibleReferenceFramesException("The two states do not have the same reference frame");
281  }
282  // calculation
283  double result = 0;
284  if (state_variable_type == CartesianStateVariable::POSITION || state_variable_type == CartesianStateVariable::POSE
285  || state_variable_type == CartesianStateVariable::ALL) {
286  result += (this->get_position() - state.get_position()).norm();
287  }
288  if (state_variable_type == CartesianStateVariable::ORIENTATION || state_variable_type == CartesianStateVariable::POSE
289  || state_variable_type == CartesianStateVariable::ALL) {
290  // https://math.stackexchange.com/questions/90081/quaternion-distance for orientation
291  double inner_product = this->get_orientation().dot(state.get_orientation());
292  double argument = 2 * inner_product * inner_product - 1;
293  result += acos(std::min(1.0, std::max(-1.0, argument)));
294  }
295  if (state_variable_type == CartesianStateVariable::LINEAR_VELOCITY
296  || state_variable_type == CartesianStateVariable::TWIST || state_variable_type == CartesianStateVariable::ALL) {
297  result += (this->get_linear_velocity() - state.get_linear_velocity()).norm();
298  }
299  if (state_variable_type == CartesianStateVariable::ANGULAR_VELOCITY
300  || state_variable_type == CartesianStateVariable::TWIST || state_variable_type == CartesianStateVariable::ALL) {
301  result += (this->get_angular_velocity() - state.get_angular_velocity()).norm();
302  }
303  if (state_variable_type == CartesianStateVariable::LINEAR_ACCELERATION
304  || state_variable_type == CartesianStateVariable::ACCELERATIONS
305  || state_variable_type == CartesianStateVariable::ALL) {
306  result += (this->get_linear_acceleration() - state.get_linear_acceleration()).norm();
307  }
308  if (state_variable_type == CartesianStateVariable::ANGULAR_ACCELERATION
309  || state_variable_type == CartesianStateVariable::ACCELERATIONS
310  || state_variable_type == CartesianStateVariable::ALL) {
311  result += (this->get_angular_acceleration() - state.get_angular_acceleration()).norm();
312  }
313  if (state_variable_type == CartesianStateVariable::FORCE || state_variable_type == CartesianStateVariable::WRENCH
314  || state_variable_type == CartesianStateVariable::ALL) {
315  result += (this->get_force() - state.get_force()).norm();
316  }
317  if (state_variable_type == CartesianStateVariable::TORQUE || state_variable_type == CartesianStateVariable::WRENCH
318  || state_variable_type == CartesianStateVariable::ALL) {
319  result += (this->get_torque() - state.get_torque()).norm();
320  }
321  return result;
322 }
