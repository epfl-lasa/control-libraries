326 void JointState::clamp_state_variable(
327  const Eigen::ArrayXd& max_absolute_value_array, const JointStateVariable& state_variable_type,
328  const Eigen::ArrayXd& noise_ratio_array
329 ) {
330  Eigen::VectorXd state_variable = this->get_state_variable(state_variable_type);
331  int expected_size = state_variable.size();
332  if (max_absolute_value_array.size() != expected_size) {
333  throw IncompatibleSizeException(
334  "Array of max values is of incorrect size: expected " + std::to_string(expected_size) + ", given "
335  + std::to_string(max_absolute_value_array.size()));
336  }
337 
338  if (noise_ratio_array.size() != expected_size) {
339  throw IncompatibleSizeException(
340  "Array of max values is of incorrect size: expected " + std::to_string(expected_size) + ", given "
341  + std::to_string(noise_ratio_array.size()));
342  }
343  for (int i = 0; i < expected_size; ++i) {
344  if (noise_ratio_array(i) != 0.0 && abs(state_variable(i)) < noise_ratio_array(i) * max_absolute_value_array(i)) {
345  // apply dead zone
346  state_variable(i) = 0.0;
347  } else if (abs(state_variable(i)) > max_absolute_value_array(i)) {
348  // clamp to max value
349  state_variable(i) *= max_absolute_value_array(i) / abs(state_variable(i));
350  }
351  }
352  this->set_state_variable(state_variable, state_variable_type);
353 }

365 double JointState::dist(const JointState& state, const JointStateVariable& state_variable_type) const {
366  // sanity check
367  if (this->is_empty()) { throw EmptyStateException(this->get_name() + " state is empty"); }
368  if (state.is_empty()) { throw EmptyStateException(state.get_name() + " state is empty"); }
369  if (!this->is_compatible(state)) {
370  throw IncompatibleStatesException(
371  "The two joint states are incompatible, check name, joint names and order or size"
372  );
373  }
374  // calculation
375  double result = 0;
376  if (state_variable_type == JointStateVariable::POSITIONS || state_variable_type == JointStateVariable::ALL) {
377  result += (this->get_positions() - state.get_positions()).norm();
378  }
379  if (state_variable_type == JointStateVariable::VELOCITIES || state_variable_type == JointStateVariable::ALL) {
380  result += (this->get_velocities() - state.get_velocities()).norm();
381  }
382  if (state_variable_type == JointStateVariable::ACCELERATIONS || state_variable_type == JointStateVariable::ALL) {
383  result += (this->get_accelerations() - state.get_accelerations()).norm();
384  }
385  if (state_variable_type == JointStateVariable::TORQUES || state_variable_type == JointStateVariable::ALL) {
386  result += (this->get_torques() - state.get_torques()).norm();
387  }
388  return result;
389 }
