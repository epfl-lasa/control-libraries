44 template<>
45 CartesianState IDynamicalSystem<CartesianState>::evaluate(const CartesianState& state) const {
46  if (this->get_base_frame().is_empty()) {
47  throw exceptions::EmptyBaseFrameException("The base frame of the dynamical system is empty.");
48  }
49  if (state.get_reference_frame() != this->get_base_frame().get_name()) {
50  if (state.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
51  throw state_representation::exceptions::IncompatibleReferenceFramesException(
52  "The evaluated state " + state.get_name() + " in frame " + state.get_reference_frame()
53  + " is incompatible with the base frame of the dynamical system " + this->get_base_frame().get_name()
54  + " in frame " + this->get_base_frame().get_reference_frame() + "."
55  );
56  }
57  CartesianState result = this->get_base_frame().inverse() * state;
58  result = this->compute_dynamics(result);
59  return this->get_base_frame() * result;
60  } else {
61  return this->compute_dynamics(state);
62  }
63 }
