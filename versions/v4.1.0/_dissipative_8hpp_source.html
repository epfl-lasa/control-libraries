236 template<class S>
237 Eigen::MatrixXd Dissipative<S>::orthonormalize_basis(const Eigen::MatrixXd& basis,
238  const Eigen::VectorXd& main_eigenvector) {
239  Eigen::MatrixXd orthonormal_basis = basis;
240  uint dim = basis.rows();
241  orthonormal_basis.col(0) = main_eigenvector.normalized();
242  for (uint i = 1; i < dim; i++) {
243  for (uint j = 0; j < i; j++) {
244  orthonormal_basis.col(i) -= orthonormal_basis.col(j).dot(orthonormal_basis.col(i)) * orthonormal_basis.col(j);
245  }
246  orthonormal_basis.col(i).normalize();
247  }
248  return orthonormal_basis;
249 }
