24 template<>
25 Eigen::MatrixXd Dissipative<CartesianState>::compute_orthonormal_basis(const CartesianState& desired_velocity) {
26  double tolerance = 1e-4;
27  Eigen::MatrixXd updated_basis = Eigen::MatrixXd::Zero(this->nb_dimensions_, this->nb_dimensions_);
28  switch (this->computational_space_) {
29  case ComputationalSpaceType::LINEAR: {
30  const Eigen::Vector3d& linear_velocity = desired_velocity.get_linear_velocity();
31  // only update the damping if the commanded linear velocity is non null
32  if (linear_velocity.norm() < tolerance) {
33  updated_basis.topLeftCorner<3, 3>() = Eigen::Matrix3d::Identity();
34  return updated_basis;
35  }
36  //return only the linear block
37  updated_basis.topLeftCorner<3, 3>() =
38  Dissipative<CartesianState>::orthonormalize_basis(this->basis_.topLeftCorner<3, 3>(), linear_velocity);
39  break;
40  }
41  case ComputationalSpaceType::ANGULAR: {
42  const Eigen::Vector3d& angular_velocity = desired_velocity.get_angular_velocity();
43  // only update the damping if the commanded angular velocity is non null
44  if (angular_velocity.norm() < tolerance) {
45  updated_basis.bottomRightCorner<3, 3>() = Eigen::Matrix3d::Identity();
46  return updated_basis;
47  }
48  // return only the angular block
49  updated_basis.bottomRightCorner<3, 3>() =
50  Dissipative<CartesianState>::orthonormalize_basis(this->basis_.bottomRightCorner<3, 3>(), angular_velocity);
51  break;
52  }
53  case ComputationalSpaceType::DECOUPLED_TWIST: {
54  // compute per block
55  bool updated = false;
56  const Eigen::Vector3d& linear_velocity = desired_velocity.get_linear_velocity();
57  const Eigen::Vector3d& angular_velocity = desired_velocity.get_angular_velocity();
58  if (linear_velocity.norm() > tolerance) {
59  updated_basis.block<3, 3>(0, 0) =
60  Dissipative<CartesianState>::orthonormalize_basis(this->basis_.topLeftCorner<3, 3>(), linear_velocity);
61  updated = true;
62  }
63  if (angular_velocity.norm() > tolerance) {
64  updated_basis.block<3, 3>(3, 3) =
65  Dissipative<CartesianState>::orthonormalize_basis(this->basis_.bottomRightCorner<3, 3>(), angular_velocity);
66  updated = true;
67  }
68  // at least the linear or angular parts have been updated
69  if (!updated) {
70  return Eigen::MatrixXd::Identity(this->nb_dimensions_, this->nb_dimensions_);
71  }
72  break;
73  }
74  case ComputationalSpaceType::FULL: {
75  // only update the damping if the commanded velocity is non null
76  if (desired_velocity.get_twist().norm() < tolerance) {
77  return Eigen::MatrixXd::Identity(this->nb_dimensions_, this->nb_dimensions_);
78  }
79  // return the full damping matrix
80  updated_basis = Dissipative<CartesianState>::orthonormalize_basis(this->basis_, desired_velocity.get_twist());
81  break;
82  }
83  }
84  return updated_basis;
85 }
