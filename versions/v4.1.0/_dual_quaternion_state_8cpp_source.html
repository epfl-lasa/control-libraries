68  // to be extra carefulm Quaternion coeffs are returned as (x,y,z,w)
69  pexps = Eigen::Quaterniond(coeffs(3) + cos(state.get_primary().norm()), coeffs(0), coeffs(1), coeffs(2));
70  } else {
