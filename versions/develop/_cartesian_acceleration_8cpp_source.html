88CartesianAcceleration& CartesianAcceleration::operator*=(const Eigen::Matrix<double, 6, 6>& lambda) {
89 // sanity check
90 if (this->is_empty()) {
91 throw EmptyStateException(this->get_name() + " state is empty");
92 }
93 // operation
94 this->set_linear_acceleration(lambda.block<3, 3>(0, 0) * this->get_linear_acceleration());
95 this->set_angular_acceleration(lambda.block<3, 3>(3, 3) * this->get_angular_acceleration());
96 return (*this);
97}

99CartesianTwist CartesianAcceleration::operator*(const std::chrono::nanoseconds& dt) const {
100 // sanity check
101 if (this->is_empty()) {
102 throw EmptyStateException(this->get_name() + " state is empty");
103 }
104 // operations
105 CartesianTwist twist(this->get_name(), this->get_reference_frame());
106 // convert the period to a double with the second as reference
107 double period = dt.count();
108 period /= 1e9;
109 // convert the acceleration into a twist
110 twist.set_linear_velocity(period * this->get_linear_acceleration());
111 twist.set_angular_velocity(period * this->get_angular_acceleration());
112 return twist;
113}
