24void Circular::set_limit_cycle(Ellipsoid& limit_cycle) {
25 const auto& center = limit_cycle.get_center_state();
26 if (center.is_empty()) {
27 throw state_representation::exceptions::EmptyStateException(center.get_name() + " state is empty");
28 }
29 if (this->get_base_frame().is_empty()) {
30 IDynamicalSystem<CartesianState>::set_base_frame(
31 CartesianState::Identity(
32 center.get_reference_frame(), center.get_reference_frame()));
33 }
34 if (center.get_reference_frame() != this->get_base_frame().get_name()) {
35 if (center.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
36 throw state_representation::exceptions::IncompatibleReferenceFramesException(
37 "The reference frame of the center " + center.get_name() + " in frame " + center.get_reference_frame()
38 + " is incompatible with the base frame of the dynamical system " + this->get_base_frame().get_name()
39 + " in frame " + this->get_base_frame().get_reference_frame() + ".");
40 }
41 limit_cycle.set_center_state(this->get_base_frame().inverse() * center);
42 }
43 this->limit_cycle_->set_value(limit_cycle);
44}

78CartesianState Circular::compute_dynamics(const CartesianState& state) const {
79 if (this->limit_cycle_->get_value().get_center_state().is_empty()) {
80 throw exceptions::EmptyAttractorException("The limit cycle of the dynamical system is empty.");
81 }
82 // put the point in the reference of the center
83 auto pose = CartesianPose(state);
84 pose = this->limit_cycle_->get_value().get_rotation().inverse()
85 * this->limit_cycle_->get_value().get_center_pose().inverse() * pose;
86
87 CartesianTwist velocity(pose.get_name(), pose.get_reference_frame());
88 Eigen::Vector3d linear_velocity;
89 linear_velocity(2) = -this->normal_gain_->get_value() * pose.get_position()(2);
90
91 std::vector<double> radiuses = this->limit_cycle_->get_value().get_axis_lengths();
92
93 double a2ratio = (pose.get_position()[0] * pose.get_position()[0]) / (radiuses[0] * radiuses[0]);
94 double b2ratio = (pose.get_position()[1] * pose.get_position()[1]) / (radiuses[1] * radiuses[1]);
95 double dradius = -this->planar_gain_->get_value() * radiuses[0] * radiuses[1] * (a2ratio + b2ratio - 1);
96 double tangent_velocity_x = -radiuses[0] / radiuses[1] * pose.get_position()[1];
97 double tangent_velocity_y = radiuses[1] / radiuses[0] * pose.get_position()[0];
98
99 linear_velocity(0) = this->circular_velocity_->get_value() * tangent_velocity_x + dradius * tangent_velocity_y;
100 linear_velocity(1) = this->circular_velocity_->get_value() * tangent_velocity_y - dradius * tangent_velocity_x;
101
102 velocity.set_linear_velocity(linear_velocity);
103 velocity.set_angular_velocity(Eigen::Vector3d::Zero());
104
105 //compute back the linear velocity in the desired frame
106 auto frame = this->limit_cycle_->get_value().get_center_pose() * this->limit_cycle_->get_value().get_rotation();
107 return CartesianState(frame) * velocity;
108}
