99 CartesianPose CartesianTwist::operator*(const std::chrono::nanoseconds& dt) const {
100  // sanity check
101  if (this->is_empty()) {
102  throw EmptyStateException(this->get_name() + " state is empty");
103  }
104  // operations
105  CartesianPose displacement(this->get_name(), this->get_reference_frame());
106  // convert the period to a double with the second as reference
107  double period = dt.count();
108  period /= 1e9;
109  // convert the velocities into a displacement
110  displacement.set_position(period * this->get_linear_velocity());
111  Eigen::Quaterniond angular_displacement = Eigen::Quaterniond::Identity();
112  double angular_norm = this->get_angular_velocity().norm();
113  if (angular_norm > 1e-4) {
114  double theta = angular_norm * period * 0.5;
115  angular_displacement.w() = cos(theta);
116  angular_displacement.vec() = this->get_angular_velocity() / angular_norm * sin(theta);
117  }
118  displacement.set_orientation(angular_displacement);
119  return displacement;
120 }
