31 template<>
32 CartesianState Impedance<CartesianState>::compute_command(const CartesianState& desired_state,
33  const CartesianState& feedback_state) {
34  CartesianState state_error = desired_state - feedback_state;
35  // compute the wrench using the formula W = I * acc_desired + K * e_pose + D * e_twist + e_wrench
36  CartesianWrench command(feedback_state.get_name(), feedback_state.get_reference_frame());
37  // compute force
38  Eigen::Vector3d position_control = this->get_stiffness().topLeftCorner<3,3>() * state_error.get_position()
39  + this->get_damping().topLeftCorner<3,3>() * state_error.get_linear_velocity()
40  + this->get_inertia().topLeftCorner<3,3>() * desired_state.get_linear_acceleration();
41  Eigen::Vector3d commanded_force = position_control + state_error.get_force();
42  command.set_force(commanded_force);
43  // compute torque (orientation requires special care)
44  Eigen::Vector3d orientation_control = this->get_stiffness().bottomRightCorner<3,3>() * state_error.get_orientation().vec()
45  + this->get_damping().bottomRightCorner<3,3>() * state_error.get_angular_velocity()
46  + this->get_inertia().bottomRightCorner<3,3>() * desired_state.get_angular_acceleration();
47  Eigen::Vector3d commanded_torque = orientation_control + state_error.get_torque();
48  command.set_torque(commanded_torque);
49  return command;
50 }

53 JointState Impedance<JointState>::compute_command(const JointState& desired_state,
54  const JointState& feedback_state) {
55  JointState state_error = desired_state - feedback_state;
56  // compute the wrench using the formula T = I * acc_desired + K * e_pos + D * e_vel + e_tor
57  JointTorques command(feedback_state.get_name(), feedback_state.get_names());
58  // compute torques
59  Eigen::VectorXd state_control = this->get_stiffness() * state_error.get_positions()
60  + this->get_damping() * state_error.get_velocities()
61  + this->get_inertia() * desired_state.get_accelerations();
62  Eigen::VectorXd commanded_torques = state_control + state_error.get_torques();
63  command.set_torques(commanded_torques);
64  return command;
65 }
